koa 日志记录

约 1646 字大约 5 分钟

2025-01-20

Node.js中可以使用log4js来记录日志。支持多种日志输出方式，如控制台、文件、数据库等。

安装

$ pnpm add -D log4js

使用

配置log4js配置

/*
 * @Author: matiastang
 * @Date: 2024-04-13 17:19:33
 * @LastEditors: matiastang
 * @LastEditTime: 2024-04-15 14:44:12
 * @FilePath: /mt-koa/src/config/logger.ts
 * @Description: log4js配置文件
 */
import path from 'path'

//日志根目录
const baseLogPath = path.resolve(process.cwd(), './logs')

/*报错输出日志*/
//错误日志目录、文件名、输出完整路径
const errorPath = "/error"
const errorFileName = "error"
const errorLogPath = baseLogPath + errorPath + "/" + errorFileName

/*请求数据得到响应时输出响应日志*/
//响应日志目录、文件名、输出完整路径
const responsePath = "/response"
const responseFileName = "response"
const responseLogPath = baseLogPath + responsePath + "/" + responseFileName

/*操作数据库进行增删改等敏感操作记录日志*/
//操作日志目录、文件名、输出完整路径
const handlePath = "/handle"
const handleFileName = "handle"
const handleLogPath = baseLogPath + handlePath + "/" + handleFileName

/**
 * 获取配置
 * 之所以是函数，是因为需要从环境变量里面获取配置
 * @returns 
 */
const getLoggerConfig = () => {

    // 服务器地址
    const EMAIL_HOST = process.env.EMAIL_HOST
    // 服务器端口
    const EMAIL_PORT = process.env.EMAIL_PORT
    // 账号
    const EMAIL_HOST_USER = process.env.EMAIL_HOST_USER
    // 密码(授权码)
    const EMAIL_HOST_PASSWORD = process.env.EMAIL_HOST_PASSWORD
    // 邮箱来自
    const EMAIL_FROM = process.env.EMAIL_FROM
    // 接收错误日志邮箱
    const ERROR_LOGGER_TO_EMAIL = process.env.ERROR_LOGGER_TO_EMAIL

    // console.log(EMAIL_HOST, EMAIL_PORT, EMAIL_HOST_USER, EMAIL_HOST_PASSWORD, EMAIL_FROM, ERROR_LOGGER_TO_EMAIL)

    return {
        // 日志格式等设置
        appenders: {
            // 默认
            console: {
                type: "console",
            },
            // 事件
            handleLogger: {
                type: "dateFile",
                filename: handleLogPath,
                pattern: "yyyy-MM-dd-hh.log",
                alwaysIncludePattern: true,
                encoding: "utf-8",
                maxLogSize: 10485760, // 10mb,日志文件大小,超过该size则自动创建新的日志文件
                numBackups: 10, // 仅保留最新的20个日志文件
                compress: true, // 超过maxLogSize,压缩代码
                path: responsePath,
            },
            // 响应配置
            responseLogger: {
                type: "dateFile",
                filename: responseLogPath,
                pattern: "yyyy-MM-dd-hh.log",
                alwaysIncludePattern: true,
                encoding: "utf-8",
                maxLogSize: 10485760, // 10mb,日志文件大小,超过该size则自动创建新的日志文件
                numBackups: 10, // 仅保留最新的20个日志文件
                compress: true, // 超过maxLogSize,压缩代码
                path: responsePath,
            },
            // 错误配置
            errorLogger: {
                type: "dateFile",
                filename: errorLogPath,
                pattern: "yyyy-MM-dd-hh.log",
                alwaysIncludePattern: true,
                encoding: "utf-8",
                maxLogSize: 10485760, // 10mb,日志文件大小,超过该size则自动创建新的日志文件
                numBackups: 10, // 仅保留最新的20个日志文件
                compress: true, // 超过maxLogSize,压缩代码
                path: errorPath,
            },
            // 邮箱日志配置
            emailLogger: {
                // 使用[@log4js-node/smtp](https://github.com/log4js-node/smtp)库发送邮件
                type: '@log4js-node/smtp',
                // 接收邮箱
                recipients: ERROR_LOGGER_TO_EMAIL,
                // 发送邮箱
                sender: EMAIL_FROM,
                // 标题
                subject: 'Log4js Mailer: mt-koa Log',
                SMTP: {
                    host: EMAIL_HOST,
                    port: EMAIL_PORT,// port - integer (optional, defaults to 25)
                    auth: {
                        user: EMAIL_HOST_USER,
                        pass: EMAIL_HOST_PASSWORD,
                    },
                },
            },
        },
        // 供外部调用的名称和对应设置定义
        categories: {
            // 默认
            default: {
                appenders: ["console"],
                level: "all",
            },
            // 事件
            handleLogger: {
                appenders: ["handleLogger"],
                level: "all",
            },
            // 响应
            responseLogger: {
                appenders: ["responseLogger"],
                level: "info",
            },
            // 请求错误
            httpErrorLogger: {
                appenders: ["errorLogger"],
                level: "error",
            },
            // 错误
            errorLogger: {
                appenders: ["errorLogger"],
                level: "error",
            },
            // 错误
            emailLogger: {
                appenders: ["emailLogger"],
                level: "info",
            },
        },
        baseLogPath: baseLogPath,
    }
}

export default getLoggerConfig

编写工具类

/*
 * @Author: matiastang
 * @Date: 2024-04-13 16:38:02
 * @LastEditors: matiastang
 * @LastEditTime: 2024-04-16 17:39:58
 * @FilePath: /mt-koa/src/utils/logger.ts
 * @Description: 日志工具
 */
import log4js from 'log4js'
import moment from 'moment'
import getLoggerConfig from '../config/logger'
import { ParameterizedContext, Request } from 'koa'

/**
 * 日志单例
 */
class Log4jsSingleton {  
    private static instance: Log4jsSingleton | null = null
    consoleLogger: log4js.Logger
    handleLogger: log4js.Logger
    responseLogger: log4js.Logger
    httpErrorLogger: log4js.Logger
    errorLogger: log4js.Logger
    emailLogger: log4js.Logger
  
    private constructor() {  
        // 私有构造函数，防止外部使用 new Log4jsSingleton() 创建实例  
    }  
  
    /**
     * 单例
     * @returns 
     */
    public static getInstance(): Log4jsSingleton {  
        if (!Log4jsSingleton.instance) {  
            const singleton = new Log4jsSingleton()
            // 加载配置文件
            log4js.configure(getLoggerConfig())
            // 获取日志记录器
            singleton.consoleLogger = log4js.getLogger()
            singleton.handleLogger = log4js.getLogger("handleLogger")
            singleton.responseLogger = log4js.getLogger("responseLogger")
            singleton.httpErrorLogger = log4js.getLogger("httpErrorLogger")
            singleton.errorLogger = log4js.getLogger("errorLogger")
            singleton.emailLogger = log4js.getLogger("emailLogger")

            Log4jsSingleton.instance = singleton
        }  
        return Log4jsSingleton.instance 
    }  
} 

/**
 * 日志格式化
 */
const formatLoggerText = {
    /**
     * 格式化-普通日志
     * @param text 
     * @returns 
     */
    info: (text: string) => {
        var logText = new String()
        //响应日志头信息
        logText += "\n" + "***************info log start ***************" + "\n"
        //响应内容
        logText += "info detail: " + "\n" + text + "\n"
        //响应日志结束信息
        logText += "*************** info log end ***************" + "\n"
        return logText
    },
    /**
     * 格式化-请求日志
     * @param request 
     * @param startTime 
     * @returns 
     */
    request: (request: Request, startTime: number) => {
        var logText = ''
        const method = request.method
        //访问方法
        logText += "request method: " + method + "\n"
        //请求原始地址
        logText += "request originalUrl:  " + request.originalUrl + "\n"
        //客户端ip
        logText += "request client ip:  " + request.ip + "\n"
        //请求参数
        if (['GET', 'HEAD'].includes(method.toUpperCase())) {
            logText += "request query:  " + JSON.stringify(request.query) + "\n"
        } else {
            logText += "request body: " + "\n" + JSON.stringify(request.body) + "\n"
        }
        //请求时间
        logText += "request time: " + moment(startTime).format('YYYY-MM-DD HH:mm:ss.SSS') + "\n"
        return logText
    },
    /**
     * 格式化-响应日志
     * @param ctx 
     * @param startTime 
     * @param processTime 
     * @returns 
     */
    response: (ctx: ParameterizedContext, startTime: number, processTime: number) => {
        var logText = ''
        //响应日志开始
        logText += "\n" + "*************** response log start ***************" + "\n"
        //添加请求日志
        logText += formatLoggerText.request(ctx.request, startTime)
        //响应耗时
        logText += "response time: " + processTime + "\n"
        //响应状态码
        logText += "response status: " + ctx.status + "\n"
        //响应内容
        logText += "response body: " + "\n" + JSON.stringify(ctx.body) + "\n"
        //响应日志结束
        logText += "*************** response log end ***************" + "\n"
        return logText
    },
    /**
     * 格式化-操作日志
     * @param text 
     * @returns 
     */
    handle: (text: string) => {
        var logText = ''
        //响应日志开始
        logText += "\n" + "***************info log start ***************" + "\n"
        //响应内容
        logText += "handle info detail: " + "\n" + text + "\n"
        //响应日志结束
        logText += "*************** info log end ***************" + "\n"
        return logText
    },
    /**
     * 格式化-请求错误日志
     * @param ctx 
     * @param err 
     * @param startTime 
     * @returns 
     */
    httpError: (ctx: ParameterizedContext, err: Error, startTime: number) => {
        var logText = ''
        //错误信息开始
        logText += "\n" + "*************** http error log start ***************" + "\n"
        //添加请求日志
        logText += formatLoggerText.request(ctx.request, startTime)
        //错误名称
        logText += "err name: " + err.name + "\n"
        //错误信息
        logText += "err message: " + err.message + "\n"
        //错误详情
        logText += "err stack: " + err.stack + "\n"
        //错误信息结束
        logText += "*************** http error log end ***************" + "\n"
        return logText
    },

    /**
     * 格式化-普通错误日志
     * @param err 
     * @returns 
     */
    error: (err: Error) => {
        var logText = ''
        //错误信息开始
        logText += "\n" + "*************** error log start ***************" + "\n"
        //错误名称
        logText += "err name: " + err.name + "\n"
        //错误信息
        logText += "err message: " + err.message + "\n"
        //错误详情
        logText += "err stack: " + err.stack + "\n"
        //错误信息结束
        logText += "*************** error log end ***************" + "\n"
        return logText
    },
}

/**
 * 普通日志
 * @param info 
 */
const logInfo = (info: string | string[] | object) => {
    const log4jsSingle = Log4jsSingleton.getInstance()
    if (typeof info === 'string') {
        log4jsSingle.consoleLogger.info(formatLoggerText.info(info))
        return
    }
    try {
        const text = JSON.stringify(info)
        log4jsSingle.consoleLogger.info(formatLoggerText.info(text))
    } catch (error) {
        if (error instanceof Error) {
            logError(error)
            return
        }
        console.error(error)
    }
}

/**
 * 响应日志
 * @param ctx 
 * @param startTime 
 * @param processTime 
 */
const logResponse = (ctx: ParameterizedContext, startTime: number, processTime: number) => {
    Log4jsSingleton.getInstance().responseLogger.info(formatLoggerText.response(ctx, startTime, processTime))
}

/**
 * 操作日志
 * @param info 
 */
const logHandle = (info: string | string[] | object) => {
    const log4jsSingle = Log4jsSingleton.getInstance()
    if (typeof info === 'string') {
        log4jsSingle.handleLogger.info(formatLoggerText.info(info))
        return
    }
    try {
        const text = JSON.stringify(info).replace(/\\n/g, "\n")
        log4jsSingle.handleLogger.info(formatLoggerText.handle(text))
    } catch (error) {
        if (error instanceof Error) {
            logError(error)
            return
        }
        console.error(error)
    }
}

/**
 * 请求错误日志
 * @param ctx 
 * @param error 
 * @param resTime 
 */
const logHttpError = (ctx: ParameterizedContext, error: Error, processTime: number) => {
    Log4jsSingleton.getInstance().httpErrorLogger.error(formatLoggerText.httpError(ctx, error, processTime))
}

/**
 * 普通错误日志
 * @param error 
 */
const logError = (error: Error) => {
    Log4jsSingleton.getInstance().errorLogger.error(formatLoggerText.error(error))
}

/**
 * 请求错误日志
 * @param ctx 
 * @param error 
 * @param resTime 
 */
const logHttpErrorEmail = (ctx: ParameterizedContext, error: Error, processTime: number) => {
    Log4jsSingleton.getInstance().emailLogger.error(formatLoggerText.httpError(ctx, error, processTime))
}

/**
 * 邮件日志
 * @param info 
 */
const logErrorEmail = (info: string | Error | object) => {
    const log4jsSingle = Log4jsSingleton.getInstance()
    if (typeof info === 'string') {
        log4jsSingle.emailLogger.info(formatLoggerText.info(info))
    }
    if (info instanceof Error) {
        log4jsSingle.emailLogger.error(formatLoggerText.error(info))
        return
    }
    try {
        const text = JSON.stringify(info)
        log4jsSingle.emailLogger.info(formatLoggerText.info(text))
    } catch (error) {
        if (error instanceof Error) {
            logError(error)
            return
        }
        console.error(error)
    }
}

export {
    Log4jsSingleton,
    logInfo,
    logHandle,
    logResponse,
    logHttpError,
    logError,
    logHttpErrorEmail,
    logErrorEmail,
}

后续就可以使用logInfo,logHandle,logResponse,logHttpError,logError,logHttpErrorEmail,logErrorEmail记录日志了。
