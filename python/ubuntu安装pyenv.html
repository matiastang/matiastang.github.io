# 手动安装

  • 下载
$ git clone https://github.com/pyenv/pyenv.git ~/.pyenv
  • 下载成功后配置zsh
echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.zshrc
echo '[[ -d $PYENV_ROOT/bin ]] && export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.zshrc
echo 'eval "$(pyenv init -)"' >> ~/.zshrc
  • 刷新zsh的配置
$ source ~/.zshrc
  • 查看版本
$ pyenv --version
pyenv 2.3.35-2-g96f93fd5
